## 🚀 応用問題1: 成績表からの情報抽出

### 問題

以下の成績データから、指定された情報を抽出してください。

```python
import pandas as pd

df = pd.DataFrame({
    "名前": ["田中", "佐藤", "鈴木", "高橋", "伊藤", "山田", "中村", "小林"],
    "クラス": ["A", "A", "B", "B", "A", "B", "A", "B"],
    "国語": [85, 72, 90, 68, 95, 78, 82, 88],
    "数学": [78, 88, 75, 92, 70, 85, 90, 72],
    "英語": [92, 80, 85, 75, 88, 70, 78, 95],
    "理科": [70, 85, 80, 88, 75, 92, 85, 78]
})
```

1. 「名前」「国語」「英語」の3列だけを抽出
2. 先頭3人の全データを抽出
3. 5人目（インデックス4）の「数学」の点数を取得
4. 最後の2人の「名前」と「理科」を抽出
5. 3人目から6人目（インデックス2〜5）の「国語」「数学」「英語」を抽出

### 基本解法（ループを使用）

```python
import pandas as pd

df = pd.DataFrame({
    "名前": ["田中", "佐藤", "鈴木", "高橋", "伊藤", "山田", "中村", "小林"],
    "クラス": ["A", "A", "B", "B", "A", "B", "A", "B"],
    "国語": [85, 72, 90, 68, 95, 78, 82, 88],
    "数学": [78, 88, 75, 92, 70, 85, 90, 72],
    "英語": [92, 80, 85, 75, 88, 70, 78, 95],
    "理科": [70, 85, 80, 88, 75, 92, 85, 78]
})

print("=== 元の DataFrame ===")
print(df)
print()

# ===== 1. 「名前」「国語」「英語」の3列だけを抽出 =====

# 基本解法：ループで列を選択
columns_to_select = ["名前", "国語", "英語"]
result1_data = {}
for col in columns_to_select:
    result1_data[col] = []
    for i in range(len(df)):
        result1_data[col].append(df.iloc[i][col])

result1 = pd.DataFrame(result1_data)

print("=== 1. 名前・国語・英語 ===")
print(result1)
print()

# ===== 2. 先頭3人の全データを抽出 =====

# 基本解法：ループで行を選択
result2_data = []
for i in range(3):
    row_data = {}
    for col in df.columns:
        row_data[col] = df.iloc[i][col]
    result2_data.append(row_data)

result2 = pd.DataFrame(result2_data)

print("=== 2. 先頭3人 ===")
print(result2)
print()

# ===== 3. 5人目（インデックス4）の「数学」の点数を取得 =====

math_score = df.iloc[4]["数学"]

print("=== 3. 5人目の数学の点数 ===")
print(f"5人目（{df.iloc[4]['名前']}）の数学: {math_score}点")
print()

# ===== 4. 最後の2人の「名前」と「理科」を抽出 =====

# 基本解法：インデックスを計算してループ
total_rows = len(df)
result4_data = {"名前": [], "理科": []}
for i in range(total_rows - 2, total_rows):
    result4_data["名前"].append(df.iloc[i]["名前"])
    result4_data["理科"].append(df.iloc[i]["理科"])

result4 = pd.DataFrame(result4_data)

print("=== 4. 最後の2人の名前と理科 ===")
print(result4)
print()

# ===== 5. 3人目から6人目の「国語」「数学」「英語」を抽出 =====

# 基本解法：範囲を指定してループ
result5_data = {"国語": [], "数学": [], "英語": []}
for i in range(2, 6):  # インデックス2〜5
    result5_data["国語"].append(df.iloc[i]["国語"])
    result5_data["数学"].append(df.iloc[i]["数学"])
    result5_data["英語"].append(df.iloc[i]["英語"])

result5 = pd.DataFrame(result5_data, index=df.iloc[2:6]["名前"].values)

print("=== 5. 3〜6人目の国語・数学・英語 ===")
print(result5)
```

## 🚀 応用問題2: 売上データの集計準備

### 問題

以下の売上データから、レポート作成に必要なデータを抽出してください。

```python
import pandas as pd

df = pd.DataFrame({
    "日付": ["2024-01-01", "2024-01-01", "2024-01-02", "2024-01-02", "2024-01-03"],
    "店舗": ["東京", "大阪", "東京", "大阪", "東京"],
    "商品": ["A", "B", "A", "A", "B"],
    "数量": [10, 5, 8, 12, 7],
    "単価": [100, 200, 100, 100, 200],
    "売上": [1000, 1000, 800, 1200, 1400]
}, index=["T001", "O001", "T002", "O002", "T003"])
```

1. インデックス "T002" の行を取得（loc 使用）
2. 「日付」「店舗」「売上」の3列のみを抽出
3. インデックス "O001" から "O002" までの行を取得（loc 使用）
4. 2行目から4行目（iloc）の「商品」「数量」「売上」を抽出
5. インデックス "T003" の「売上」の値を 1500 に変更

### 基本解法（ループを使用）

```python
import pandas as pd

df = pd.DataFrame({
    "日付": ["2024-01-01", "2024-01-01", "2024-01-02", "2024-01-02", "2024-01-03"],
    "店舗": ["東京", "大阪", "東京", "大阪", "東京"],
    "商品": ["A", "B", "A", "A", "B"],
    "数量": [10, 5, 8, 12, 7],
    "単価": [100, 200, 100, 100, 200],
    "売上": [1000, 1000, 800, 1200, 1400]
}, index=["T001", "O001", "T002", "O002", "T003"])

print("=== 元の DataFrame ===")
print(df)
print()

# ===== 1. インデックス "T002" の行を取得 =====

print("=== 1. インデックス T002 の行 ===")
print(row_t002)
print()

# ===== 2. 「日付」「店舗」「売上」の3列のみを抽出 =====

columns_to_select = ["日付", "店舗", "売上"]
result2_data = {}
for col in columns_to_select:
    result2_data[col] = []
    for idx in df.index:
        result2_data[col].append(df.loc[idx, col])

result2 = pd.DataFrame(result2_data, index=df.index)

print("=== 2. 日付・店舗・売上 ===")
print(result2)
print()

# ===== 3. インデックス "O001" から "O002" までの行を取得 =====

result3 = df.loc["O001":"O002"]

print("=== 3. O001 から O002 まで ===")
print(result3)
print()

# ===== 4. 2行目から4行目の「商品」「数量」「売上」を抽出 =====

# 基本解法：iloc でインデックスを指定してループ
result4_data = {"商品": [], "数量": [], "売上": []}
for i in range(1, 4):  # 2行目=index1, 4行目=index3
    result4_data["商品"].append(df.iloc[i]["商品"])
    result4_data["数量"].append(df.iloc[i]["数量"])
    result4_data["売上"].append(df.iloc[i]["売上"])

result4 = pd.DataFrame(result4_data)

print("=== 4. 2〜4行目の商品・数量・売上 ===")
print(result4)
print()

# ===== 5. インデックス "T003" の「売上」の値を 1500 に変更 =====

print("=== 5. T003 の売上を変更 ===")
print(f"変更前: {df.loc['T003', '売上']}")
df.at["T003", "売上"] = 1500
print(f"変更後: {df.loc['T003', '売上']}")
print()

print("=== 変更後の DataFrame ===")
print(df)
```

## 🚀 応用問題3: データの部分更新

### 問題

以下の在庫データに対して、指定された更新を行ってください。

```python
import pandas as pd

df = pd.DataFrame({
    "商品名": ["ノートPC", "マウス", "キーボード", "モニター", "USBメモリ"],
    "カテゴリ": ["PC", "周辺機器", "周辺機器", "PC", "周辺機器"],
    "価格": [80000, 2000, 5000, 30000, 1000],
    "在庫": [10, 50, 30, 15, 100],
    "状態": ["販売中", "販売中", "販売中", "販売中", "販売中"]
})
```

1. 「USBメモリ」（最後の行）の在庫を 80 に更新
2. 「モニター」（3行目）の価格を 28000 に更新
3. 先頭2商品の「状態」を「セール中」に更新
4. 全商品の在庫を10増やす（新しい「更新後在庫」列として追加）
5. 更新後の DataFrame を表示

<details>
<summary>解答例を見る</summary>

### 基本解法（ループを使用）

```python
import pandas as pd

df = pd.DataFrame({
    "商品名": ["ノートPC", "マウス", "キーボード", "モニター", "USBメモリ"],
    "カテゴリ": ["PC", "周辺機器", "周辺機器", "PC", "周辺機器"],
    "価格": [80000, 2000, 5000, 30000, 1000],
    "在庫": [10, 50, 30, 15, 100],
    "状態": ["販売中", "販売中", "販売中", "販売中", "販売中"]
})

print("=== 元の DataFrame ===")
print(df)
print()

# ===== 1. 「USBメモリ」（最後の行）の在庫を 80 に更新 =====

print("=== 1. USBメモリの在庫を更新 ===")
print(f"変更前: {df.iat[-1, 3]}")
df.iat[-1, 3] = 80  # 最後の行の在庫列（3列目）
print(f"変更後: {df.iat[-1, 3]}")
print()

# ===== 2. 「モニター」（3行目）の価格を 28000 に更新 =====

print("=== 2. モニターの価格を更新 ===")
print(f"変更前: {df.iloc[3]['価格']}")
df.iat[3, 2] = 28000  # 3行目の価格列（2列目）
print(f"変更後: {df.iloc[3]['価格']}")
print()

# ===== 3. 先頭2商品の「状態」を「セール中」に更新 =====

print("=== 3. 先頭2商品の状態を更新 ===")
for i in range(2):
    print(f"{df.iloc[i]['商品名']}: {df.iloc[i]['状態']} → セール中")
    df.iat[i, 4] = "セール中"  # 状態列（4列目）
print()

# ===== 4. 全商品の在庫を10増やす =====

print("=== 4. 更新後在庫列を追加 ===")
updated_stock = []
for i in range(len(df)):
    new_stock = df.iloc[i]["在庫"] + 10
    updated_stock.append(new_stock)

df["更新後在庫"] = updated_stock

print("在庫 + 10 = 更新後在庫")
for i in range(len(df)):
    print(f"  {df.iloc[i]['商品名']}: {df.iloc[i]['在庫']} + 10 = {df.iloc[i]['更新後在庫']}")
print()

# ===== 5. 更新後の DataFrame を表示 =====

print("=== 5. 更新後の DataFrame ===")
print(df)
```

## 🚀 応用問題4: 複雑な選択と集計

### 問題

以下のデータから、複数の条件を組み合わせてデータを抽出し、集計してください。

```python
import pandas as pd

df = pd.DataFrame({
    "社員ID": ["E001", "E002", "E003", "E004", "E005", "E006"],
    "名前": ["田中", "佐藤", "鈴木", "高橋", "伊藤", "山田"],
    "部署": ["営業", "開発", "営業", "人事", "開発", "営業"],
    "入社年": [2018, 2020, 2019, 2021, 2018, 2022],
    "基本給": [300000, 280000, 320000, 260000, 350000, 250000],
    "賞与": [100000, 80000, 120000, 60000, 150000, 50000]
})
```

1. 社員ID をインデックスに設定
2. 「E002」から「E005」までの社員の「名前」「部署」「基本給」を取得
3. 偶数行（0, 2, 4行目）の全データを取得
4. 全社員の「年収」（基本給×12 + 賞与）を計算して新しい列として追加
5. 「E001」と「E006」の年収を比較して表示

<details>
<summary>解答例を見る</summary>

### 基本解法（ループを使用）

```python
import pandas as pd

df = pd.DataFrame({
    "社員ID": ["E001", "E002", "E003", "E004", "E005", "E006"],
    "名前": ["田中", "佐藤", "鈴木", "高橋", "伊藤", "山田"],
    "部署": ["営業", "開発", "営業", "人事", "開発", "営業"],
    "入社年": [2018, 2020, 2019, 2021, 2018, 2022],
    "基本給": [300000, 280000, 320000, 260000, 350000, 250000],
    "賞与": [100000, 80000, 120000, 60000, 150000, 50000]
})

print("=== 元の DataFrame ===")
print(df)
print()

# ===== 1. 社員ID をインデックスに設定 =====

df = df.set_index("社員ID")

print("=== 1. 社員IDをインデックスに設定 ===")
print(df)
print()

# ===== 2. 「E002」から「E005」までの「名前」「部署」「基本給」を取得 =====

# ===== 3. 偶数行（0, 2, 4行目）の全データを取得 =====

# 基本解法：ループで偶数インデックスを収集
even_rows = []
for i in range(len(df)):
    if i % 2 == 0:  # 偶数かどうか
        even_rows.append(i)

print(result3)
print()

# ===== 4. 全社員の「年収」を計算して追加 =====

# 基本解法：ループで計算
annual_salaries = []
for idx in df.index:
    base = df.loc[idx, "基本給"]
    bonus = df.loc[idx, "賞与"]
    annual = base * 12 + bonus
    annual_salaries.append(annual)

df["年収"] = annual_salaries

print("=== 4. 年収列を追加 ===")
print(df)
print()

# ===== 5. 「E001」と「E006」の年収を比較 =====

salary_e001 = df.loc["E001", "年収"]
salary_e006 = df.loc["E006", "年収"]

print("=== 5. E001 と E006 の年収比較 ===")

print(f"E006（{df.loc['E006', '名前']}）: {salary_e006:,}円")
print(f"差額: {salary_e001 - salary_e006:,}円")

if salary_e001 > salary_e006:
    print(f"→ {df.loc['E001', '名前']}の方が高い")
elif salary_e001 < salary_e006:
    print(f"→ {df.loc['E006', '名前']}の方が高い")
else:
    print("→ 同じ年収")
```
